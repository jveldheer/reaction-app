  <div id="countdown"></div>


  <script>

    // Create or resume the AudioContext (required for mobile devices)

    const audioContext = new (window.AudioContext || window.webkitAudioContext)();


    // Function to play a beep tone (used in the countdown)

    function beep(duration, frequency, volume, type) {

      const oscillator = audioContext.createOscillator();

      const gainNode = audioContext.createGain();

      oscillator.connect(gainNode);

      gainNode.connect(audioContext.destination);

      oscillator.frequency.value = frequency || 1000;

      oscillator.type = type || "sine";

      gainNode.gain.value = volume || 1;

      oscillator.start();

      setTimeout(() => oscillator.stop(), duration);

    }


    // Function to play a simple clapping sound using white noise and an envelope

    function playClap() {

      const ctx = audioContext;

      const bufferDuration = 0.2; // seconds

      const bufferSize = ctx.sampleRate * bufferDuration;

      const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);

      const data = buffer.getChannelData(0);

      for (let i = 0; i < bufferSize; i++) {

        data[i] = Math.random() * 2 - 1;

      }

      const noiseSource = ctx.createBufferSource();

      noiseSource.buffer = buffer;


      const gainNode = ctx.createGain();

      // Set an envelope for a quick decay

      gainNode.gain.setValueAtTime(1, ctx.currentTime);

      gainNode.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + bufferDuration);


      noiseSource.connect(gainNode);

      gainNode.connect(ctx.destination);

      noiseSource.start();

    }


    // Recursive function to schedule clapping sounds until the drill duration is over

    function scheduleClap(endTime) {

      if (Date.now() < endTime) {

        const randomDelay = Math.random() * 2000 + 1000; // 1000ms to 3000ms

        setTimeout(() => {

          playClap();

          scheduleClap(endTime);

        }, randomDelay);

      }

    }


    // Start button event listener

    document.getElementById('startButton').addEventListener('click', () => {

      // Resume AudioContext on user interaction (needed on many mobile devices)

      audioContext.resume();

      const durationInput = document.getElementById('duration');

      const totalDuration = parseInt(durationInput.value) * 1000; // convert seconds to ms

      if (isNaN(totalDuration) || totalDuration <= 0) {

        alert('Please enter a valid duration in seconds.');

        return;

      }


      // Disable controls during the drill

      durationInput.disabled = true;

      document.getElementById('startButton').disabled = true;

      

      const countdownElement = document.getElementById('countdown');

      let count = 3;


      // Countdown function with beeps each second

      function countdown() {

        if (count > 0) {

          countdownElement.innerText = count;

          beep(200, 1000, 1, "sine");

          count--;

          setTimeout(countdown, 1000);

        } else {

          countdownElement.innerText = "Go!";

          setTimeout(() => {

            // Start the clapping drill

            const drillEndTime = Date.now() + totalDuration;

            scheduleClap(drillEndTime);

            countdownElement.innerText = ""; // Clear the countdown display


            // Re-enable controls after the drill ends

            setTimeout(() => {

              durationInput.disabled = false;

              document.getElementById('startButton').disabled = false;

            }, totalDuration);

          }, 500);

        }

      }

      

      countdown();

    });

  </script>
